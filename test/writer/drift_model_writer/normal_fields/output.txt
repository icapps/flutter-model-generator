import 'package:drift/drift.dart';
import 'package:model_generator_example/base_dir/path_to_my_model/person.dart';
import 'package:model_generator_example/database/model_generator_example_database.dart';

@DataClassName('DbPerson')
class DbPersonTable extends Table {
  TextColumn get firstName => text()();

  BoolColumn get boolField => boolean()();

  DateTimeColumn get birthDay => dateTime()();

  RealColumn get doubleField => real()();

  IntColumn get intField => integer()();

}

extension DbPersonExtension on DbPerson {
  Person get model => Person(
        firstName: firstName,
        boolField: boolField,
        birthDay: birthDay,
        doubleField: doubleField,
        intField: intField,
      );
}

extension PersonExtension on Person {
  DbPerson get dbModel => DbPerson(
        firstName: firstName,
        boolField: boolField,
        birthDay: birthDay,
        doubleField: doubleField,
        intField: intField,
      );
}
